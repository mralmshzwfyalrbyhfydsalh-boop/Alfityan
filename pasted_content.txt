// ================== تطبيق الفتيان v3 - آمن ومُحسَّن ==================
const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const Database = require('better-sqlite3');
const bcrypt = require('bcryptjs');
const crypto = require('crypto');

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  maxHttpBufferSize: 1e4 // حد أقصى للبيانات 10KB
});
const db = new Database('alfityan.db');

// إعدادات
const MAX_USERNAME = 20;
const MAX_PASSWORD = 50;
const MAX_MESSAGE = 1000;
const MSG_COOLDOWN = 500; // مللي ثانية بين الرسائل

// إعداد قاعدة البيانات
db.exec(`
  CREATE TABLE IF NOT EXISTS users (
    username TEXT PRIMARY KEY,
    password TEXT NOT NULL
  );
  CREATE TABLE IF NOT EXISTS messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    fromUser TEXT NOT NULL,
    toUser TEXT NOT NULL,
    text TEXT NOT NULL,
    time INTEGER NOT NULL
  );
  CREATE INDEX IF NOT EXISTS idx_messages_users ON messages(fromUser, toUser);
`);

// تنظيف الرسائل القديمة (أقدم من 24 ساعة)
function cleanOldMessages() {
  const cutoff = Date.now() - 86400000;
  db.prepare('DELETE FROM messages WHERE time < ?').run(cutoff);
}
cleanOldMessages();
setInterval(cleanOldMessages, 60 * 60 * 1000);

// المستخدمون المتصلون: socket.id -> username
const onlineUsers = new Map();
// تتبع آخر وقت إرسال لكل مستخدم لمنع الإغراق
const lastMsgTime = new Map();

// دالة مساعدة: التحقق من اسم المستخدم
function isValidUsername(str) {
  return typeof str === 'string' && str.length >= 2 && str.length <= MAX_USERNAME && /^[\u0600-\u06FF\w]+$/.test(str);
}

// ========== WebSocket Logic ==========
io.on('connection', (socket) => {
  console.log('✅ جهاز متصل:', socket.id);

  // تسجيل الدخول / إنشاء حساب
  socket.on('login', async ({ username, password }) => {
    // التحقق من المدخلات
    if (!isValidUsername(username) || !password || password.length > MAX_PASSWORD) {
      return socket.emit('loginError', 'الاسم أو كلمة المرور غير صالحة');
    }

    let user = db.prepare('SELECT * FROM users WHERE username = ?').get(username);
    try {
      if (user) {
        // تحقق من كلمة المرور
        const match = await bcrypt.compare(password, user.password);
        if (!match) return socket.emit('loginError', 'كلمة المرور خاطئة');
      } else {
        // إنشاء مستخدم جديد
        const hashed = await bcrypt.hash(password, 10);
        db.prepare('INSERT INTO users (username, password) VALUES (?, ?)').run(username, hashed);
      }
    } catch (err) {
      return socket.emit('loginError', 'خطأ في الخادم');
    }

    // إزالة أي اتصال سابق بنفس الاسم
    for (const [id, name] of onlineUsers.entries()) {
      if (name === username) {
        onlineUsers.delete(id);
        io.to(id).emit('forceLogout', 'تم تسجيل الدخول من مكان آخر');
        io.sockets.sockets.get(id)?.disconnect();
        break;
      }
    }

    // ربط المقبس الحالي بالمستخدم
    onlineUsers.set(socket.id, username);
    socket.username = username;
    socket.join(username); // غرفة خاصة

    socket.emit('loginSuccess', username);
    io.emit('updateOnlineUsers', Array.from(onlineUsers.values()));
    console.log(`👤 ${username} سجل الدخول`);
  });

  // طلب كل المستخدمين
  socket.on('getAllUsers', () => {
    const users = db.prepare('SELECT username FROM users').all().map(u => u.username);
    socket.emit('allUsers', users);
  });

  // فتح محادثة
  socket.on('getChat', (partner) => {
    const user = socket.username;
    if (!user || !partner || typeof partner !== 'string') return;
    // منع محاولة فتح محادثة مع النفس
    if (partner === user) return;

    const msgs = db.prepare(`
      SELECT fromUser, text, time FROM messages
      WHERE (fromUser = ? AND toUser = ?) OR (fromUser = ? AND toUser = ?)
      ORDER BY time ASC LIMIT 200
    `).all(user, partner, partner, user);
    socket.emit('chatHistory', msgs);
  });

  // إرسال رسالة
  socket.on('sendMessage', ({ to, text }) => {
    const from = socket.username;
    if (!from || !to || !text || typeof text !== 'string') return;
    if (text.trim().length === 0 || text.length > MAX_MESSAGE) return;
    if (to === from) return;

    // Rate limiting
    const now = Date.now();
    const last = lastMsgTime.get(from) || 0;
    if (now - last < MSG_COOLDOWN) {
      return socket.emit('msgTooFast', 'تمهل قليلاً بين الرسائل');
    }
    lastMsgTime.set(from, now);

    const time = now;
    db.prepare('INSERT INTO messages (fromUser, toUser, text, time) VALUES (?, ?, ?, ?)').run(from, to, text, time);
    const message = { from, text, time };

    // إرسال للمرسل
    socket.emit('newMessage', message);
    // إرسال للمستقبل إن كان متصلاً
    io.to(to).emit('newMessage', message);
  });

  // حذف يدوي للرسائل القديمة
  socket.on('deleteOldMessages', () => {
    const user = socket.username;
    if (!user) return;
    const cutoff = Date.now() - 86400000;
    db.prepare('DELETE FROM messages WHERE (fromUser = ? OR toUser = ?) AND time < ?').run(user, user, cutoff);
    socket.emit('oldMessagesDeleted');
  });

  // قطع الاتصال
  socket.on('disconnect', () => {
    if (socket.username) {
      onlineUsers.delete(socket.id);
      io.emit('updateOnlineUsers', Array.from(onlineUsers.values()));
      console.log(`❌ ${socket.username} خرج`);
    }
  });

  // إعادة المصادقة تلقائياً (إذا كان قد سجل الدخول سابقاً)
  socket.on('reAuth', (username) => {
    // لا نسمح بإعادة المصادقة من العميل بدون كلمة مرور
    // بدلاً من ذلك، يجب أن يحتفظ العميل برمز، لكننا سنبسطها:
    // لن نقبل reAuth؛ عند انقطاع الاتصال يجب على العميل إرسال login مرة أخرى.
  });
});

<body>
<div class="app" id="app">
  <div id="loginScreen" class="screen active login-screen">
    <h2>💬 الفتيان</h2>
    <input id="loginUser" placeholder="اسم المستخدم" autocomplete="off" maxlength="20">
    <input id="loginPass" type="password" placeholder="كلمة المرور" maxlength="50">

  <div id="mainScreen" class="screen">
    <div class="header">

      <button style="background:#c0392b;color:#fff;border:none;padding:8px 16px;border-radius:20px;" onclick="logout()">خروج</button>

    <div class="contacts" id="contactsContainer"></div>

  <div id="chatScreen" class="screen">
    <div class="chat-header">
      <button style="background:none;border:none;color:#fff;font-size:1.4rem;" onclick="closeChat()">←</button>
      <span id="chatPartnerName"></span>
    </div>
    <div class="messages" id="messagesContainer"></div>
    <div class="input-area">
      <input id="messageInput" placeholder="اكتب رسالة..." autocomplete="off" maxlength="1000">
      <button onclick="send()">➤</button>
    </div>
  </div>

<script>
  const socket = io({ reconnection: true, reconnectionAttempts: 10 });
  let currentUser = null;
  let currentChat = null;

  // ---------- وظائف الشاشات ----------
  function showScreen(id) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById(id).classList.add('active');
  }

  // ---------- أحداث الشبكة ----------
  socket.on('loginError', msg => document.getElementById('loginError').textContent = msg);
  socket.on('loginSuccess', username => {
    currentUser = username;
    document.getElementById('myNameDisplay').textContent = username;
    document.getElementById('loginError').textContent = '';
    showScreen('mainScreen');
    socket.emit('getAllUsers');
  });

  socket.on('forceLogout', (msg) => {
    alert(msg);
    logout();
  });

  socket.on('updateOnlineUsers', online => {
    window._online = online || [];
    renderContacts();
  });

  socket.on('allUsers', users => {
    window._allUsers = users || [];
    renderContacts();
  });

  // عرض جهات الاتصال بشكل آمن (بدون innerHTML)
  function renderContacts() {
    const all = window._allUsers || [];
    const online = window._online || [];
    const container = document.getElementById('contactsContainer');
    container.innerHTML = ''; // مسح

    all.forEach(user => {
      if (user === currentUser) return;
      const isOnline = online.includes(user);

      const contactDiv = document.createElement('div');
      contactDiv.className = 'contact';
      contactDiv.onclick = () => startChat(user);

      const avatar = document.createElement('div');
      avatar.className = 'avatar';
      avatar.textContent = user.charAt(0);

      const nameDiv = document.createElement('div');
      nameDiv.className = 'name';
      nameDiv.textContent = user;

      const dot = document.createElement('div');
      dot.className = 'online-dot';
      dot.style.background = isOnline ? '#2ecc71' : '#cbd5e1';

      contactDiv.appendChild(avatar);
      contactDiv.appendChild(nameDiv);
      contactDiv.appendChild(dot);
      container.appendChild(contactDiv);
    });
  }

  // ---------- المحادثة ----------
  function startChat(partner) {
    currentChat = partner;
    document.getElementById('chatPartnerName').textContent = partner;
    showScreen('chatScreen');
    document.getElementById('messagesContainer').innerHTML = '';
    socket.emit('getChat', partner);
  }
  function closeChat() {
    currentChat = null;
    showScreen('mainScreen');
    socket.emit('getAllUsers');
  }

  socket.on('chatHistory', msgs => {
    const container = document.getElementById('messagesContainer');
    container.innerHTML = '';
    msgs.forEach(msg => displayMsg(msg));
    container.scrollTop = container.scrollHeight;
  });

  socket.on('newMessage', msg => {
    if (currentChat && (msg.from === currentChat || msg.from === currentUser)) {
      displayMsg(msg);
    }
    socket.emit('getAllUsers'); // لتحديث القائمة
  });

  function displayMsg(msg) {
    const container = document.getElementById('messagesContainer');
    const div = document.createElement('div');
    div.className = \`msg \${msg.from === currentUser ? 'sent' : 'received'}\`;
    div.textContent = msg.text; // آمن
    const timeSpan = document.createElement('span');
    timeSpan.className = 'time';
    timeSpan.textContent = new Date(msg.time).toLocaleTimeString('ar-SA', { hour:'2-digit', minute:'2-digit' });
    div.appendChild(timeSpan);
    container.appendChild(div);
    container.scrollTop = container.scrollHeight;
  }

  function send() {
    const input = document.getElementById('messageInput');
    const text = input.value.trim();
    if (!text || !currentChat) return;
    socket.emit('sendMessage', { to: currentChat, text });
    input.value = '';
  }

  // تنبيه السرعة
  socket.on('msgTooFast', msg => alert(msg));

  // ---------- تسجيل الدخول / الخروج ----------
  function login() {
    const u = document.getElementById('loginUser').value.trim();
    const p = document.getElementById('loginPass').value.trim();
    if (!u || !p) {
      document.getElementById('loginError').textContent = 'املأ الحقول';
      return;
    }
    socket.emit('login', { username: u, password: p });
  }

  function logout() {
    currentUser = null;
    currentChat = null;
    showScreen('loginScreen');
    socket.disconnect();
    // إعادة تحميل المقبس (سيبدأ جديدًا)
    setTimeout(() => socket.connect(), 300);
  }

  // ---------- حذف القديم ----------
  function deleteOld() {
    if (confirm('حذف كل الرسائل التي مضى عليها أكثر من يوم؟')) {
      socket.emit('deleteOldMessages');
    }
  }
  socket.on('oldMessagesDeleted', () => alert('تم الحذف'));

  // ---------- مؤشر انقطاع الاتصال ----------
  socket.on('disconnect', () => {
    document.getElementById('disconnectedAlert').style.display = 'block';
  });
  socket.on('connect', () => {
    document.getElementById('disconnectedAlert').style.display = 'none';
    // عند إعادة الاتصال، لا نفقد المستخدم إذا كان لا يزال مسجلاً؟ نطلب منه إعادة login

      alert('انقطع الاتصال، الرجاء تسجيل الدخول مرة أخرى.');
      logout();
    }
  });

  // دعم Enter
  document.getElementById('messageInput')?.addEventListener('keypress', e => { if (e.key === 'Enter') send(); });
  document.getElementById('loginPass')?.addEventListener('keypress', e => { if (e.key === 'Enter') login(); });
</script>
</body>
</html>
  `);
});
